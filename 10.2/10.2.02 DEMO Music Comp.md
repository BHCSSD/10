function youSpin() {
    push();
    angleMode(DEGREES); // JavaScript uses radians by default; switching to degrees
    translate(300, 150); // Move origin to center
    rotate(angle); // Spin

    angle++;
    if (angle > 360) {
        angle = 0;
    }

    imageMode(CENTER);
    // image(logo, 0, 0); // optional: add your own logo
